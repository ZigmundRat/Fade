ConfigLed(0, "RGB", 33, 13)
FUNC Scan(red, green, blue)
  for channel 0:31
    Di(1,channel,{red, green, blue})
    D(10,channel,{0, 0, 0})
    A(3)
  endfor
  for channel 32:1:-1
    Di(1,channel,{red, green, blue})
    D(10,channel,{0, 0, 0})
    A(3)
  endfor          
ENDFUNC

Scan(1, 0, 0)
Scan(0, 1, 0)
Scan(0, 0, 1)       
Scan(1, 1, 0)
Scan(0, 1, 1)
Scan(1, 0, 1)
