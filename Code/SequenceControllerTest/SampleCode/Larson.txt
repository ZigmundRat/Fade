CONFIGLED(0, "RGB", 33, 13)
FUNC Scan(red, green, blue)
  FOR channel 0:31
    DI(1,channel,{red, green, blue})
    D(10,channel,{0, 0, 0})
    A(3)
  ENDFOR
  FOR channel 32:1:-1
    DI(1,channel,{red, green, blue})
    D(10,channel,{0, 0, 0})
    A(3)
  ENDFOR          
ENDFUNC

Scan(1, 0, 0)
Scan(0, 1, 0)
Scan(0, 0, 1)       
Scan(1, 1, 0)
Scan(0, 1, 1)
Scan(1, 0, 1)
